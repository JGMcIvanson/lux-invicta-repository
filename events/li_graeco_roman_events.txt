character_event = {
	id = vestalmaint.1
	

	# hide_window = yes
	is_triggered_only = yes
	

	
	trigger = { 
		has_landed_title = c_vesta_temple
	}
	
	immediate = {
		if = {
			limit = { NOT = { has_minor_title = title_vestalis_maxima } }
			random_courtier = {
				limit = {
					NOT = { character = ROOT }
					has_minor_title = title_vestalis
					trait = vestalis
					is_female = yes
				}
				c_vesta_temple = { gain_title = PREV }
				b_vesta_temple = { gain_title = PREV }

				remove_title = title_vestalis
				give_minor_title = title_vestalis_maxima
				ROOT = { set_character_flag = vestalis_maxima_found }
			}
			if = {
				limit = { NOT = { has_character_flag = vestalis_maxima_found } }
				random_list = {
					10 = { create_random_priest = {
						random_traits = yes
						religion = roman_imperial
						culture = roman
						trait = vestalis
						female = yes
						age = 30
					} }
					10 = { create_random_priest = {
						random_traits = yes
						religion = roman_imperial
						culture = roman
						trait = vestalis
						female = yes
						age = 35
					} }
					10 = { create_random_priest = {
						random_traits = yes
						religion = roman_imperial
						culture = roman
						trait = vestalis
						female = yes
						age = 40
					} }
					20 = { create_random_priest = {
						random_traits = yes
						religion = roman_imperial
						culture = roman
						trait = vestalis
						female = yes
						age = 45
					} }
					20 = { create_random_priest = {
						random_traits = yes
						religion = roman_imperial
						culture = roman
						trait = vestalis
						female = yes
						age = 50
					} }
					20 = { create_random_priest = {
						random_traits = yes
						religion = roman_imperial
						culture = roman
						trait = vestalis
						female = yes
						age = 55

						random_traits = yes
						religion = roman_imperial

						female = yes

		name = OK 
	}

	id = ancrel.2124

	desc = EVTDESCancrel.2124
	picture = GFX_evt_princess_byzantine

	is_triggered_only = yes

	option = {
		name = EVTOPTAancrel.2124
		ai_chance = {
			factor = 60
			modifier = {
				factor = 0
				OR = {
					is_primary_heir = yes
					is_pretender = yes
				}
			}
			modifier = {
				factor = 1.5
				dynasty = ROOT
			}
			modifier = {
				factor = 0.8
				NOT = { FROM = { host = { reverse_opinion = { who = ROOT value = -40 } } } }
			}
			modifier = {
				factor = 1.25
				FROM = { host = { reverse_opinion = { who = ROOT value = 40 } } }
			}
			modifier = {
				factor = 0.5
				check_variable = { which = "num_vestales" value = 4 }
			}
			modifier = {
				factor = 2
				NOT = { check_variable = { which = "num_vestales" value = 2 } }
			}
		}

		FROM = { clr_character_flag = flag_nominated_as_vestalis }
		hidden_tooltip = { FROM = { banish = ROOT } } # Move FROM to ROOT's court
		FROM = {
			host = { prestige = 50 }
			host = { piety = 50 }
			host = { opinion = { modifier = opinion_nominated_vestalis who = ROOT } }
		}

		FROM = {
			give_minor_title = title_vestalis
			hidden_tooltip = {
				remove_trait = hedonist
				add_trait = celibate
				add_trait = vestalis
				recalc_succession = yes
				set_character_flag = flag_vestalis
				c_vesta_temple = { owner = { change_variable = { which = "num_vestales" value = 1 } } }
			}
		}
	}

	option = {
		name = EVTOPTBancrel.2124
		ai_chance = {
			factor = 40
		}

		FROM = { host = { prestige = -10 } }
	}
}
